    <input id="url" placeholder="Paste YouTube or video URL" />
    <button id="convert">Convert to MP3</button>
    <a id="download" hidden>Download MP3</a>
    <div class="note">Requires backend (Node.js + yt-dlp)</div>
  </div>

  <script>
    const btn = document.getElementById('convert');
    const input = document.getElementById('url');
    const link = document.getElementById('download');

    btn.onclick = async () => {
      btn.disabled = true;
      btn.textContent = 'Converting...';

      const res = await fetch('/convert', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ url: input.value })
      });

      const data = await res.json();
      link.href = data.file;
      link.hidden = false;

      btn.disabled = false;
      btn.textContent = 'Convert to MP3';
    };
  </script>
